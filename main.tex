% Slide 2: Motivation
\begin{frame}{From Intuition to Formalization}
    \textbf{Core Question:} How does \remph{a single training data point} affect our model's predictions?
    
    \vspace{0.5cm}
    Imagine a scenario where we want to understand a model's behavior, find mislabeled data, or defend against data poisoning attacks.
    
    \pause
    \vspace{0.5cm}
    \textbf{The Naive Approach (\emph{Leave-One-Out} (LOO) Retraining):}
    \begin{itemize}
        \item Remove the target data point $z$ from the training set.
        \item Retrain the model entirely from scratch.
        \item Compare the new parameters or predictions with the original ones.
    \end{itemize}
    
    \pause
    \vspace{0.5cm}
    \textbf{The Problem:} %Retraining a modern neural network for every single training point is 
    LOO is \emph{computationally impossible}. We need an efficient approximation.
\end{frame}

% Slide 3: Empirical Risk Minimization
\begin{frame}{Mathematical Setup}
    Let $z = (x, y)$ be a training point, $L(z, \theta)$ the loss, and $\hat{\theta}$ the ERM solution:

    \begin{equation}
        \hat{\theta} = \arg\min_{\theta} \frac{1}{n} \sum_{i=1}^{n} L(z_i, \theta) = \arg\min_{\theta} R(\theta)
    \end{equation}

    \pause

    \vspace{0.5cm}
    Upweight a single training point $z$ by $\epsilon$:

    \begin{equation}
        \hat{\theta}_{\epsilon, z} = \arg\min_{\theta} \eqnmarkbox[blue]{erm}{R(\theta)} + \eqnmarkbox[red]{upw}{\epsilon \, L(z, \theta)}
    \end{equation}
    \annotate[yshift=-1em]{below,left}{erm}{\small empirical risk}
    \annotate[yshift=-1em]{below,right}{upw}{\small extra weight $\epsilon$ on $z$}

\end{frame}

% Slide 4a: Deriving the Influence Function (Part 1)
\begin{frame}{Deriving the Influence Function (I)}
    \textbf{Goal:} compute how $\hat{\theta}_{\epsilon, z}$ changes as we vary $\epsilon$, i.e.\ $\left.\dfrac{d\hat{\theta}_{\epsilon,z}}{d\epsilon}\right|_{\epsilon=0}$.

    \vspace{0.5cm}
    \textbf{Step 1.} Stationarity condition --- $\hat{\theta}_{\epsilon, z}$ minimizes the upweighted objective:
    \begin{equation}
        \nabla_\theta R(\hat{\theta}_{\epsilon,z}) + \epsilon\,\nabla_\theta L(z, \hat{\theta}_{\epsilon,z}) = 0
    \end{equation}

    \pause
    \vspace{0.3cm}
    \textbf{Step 2.} Taylor expansion of $\nabla_\theta R(\hat{\theta}_{\epsilon,z})$ around $\hat{\theta}$ (at $\epsilon=0$):
    \[
        \nabla_\theta R(\hat{\theta}_{\epsilon,z}) \approx \underbrace{\nabla_\theta R(\hat{\theta})}_{=\,0} + \nabla_\theta^2 R(\hat{\theta})(\hat{\theta}_{\epsilon,z} - \hat{\theta})
    \]
\end{frame}

% Slide 4b: Deriving the Influence Function (Part 2)
\begin{frame}{Deriving the Influence Function (II)}
    \textbf{Step 3.} Substitute into the stationarity condition:
    \[
        \underbrace{\nabla_\theta^2 R(\hat{\theta})}_{H_{\hat{\theta}}} (\hat{\theta}_{\epsilon,z} - \hat{\theta}) + \epsilon\,\nabla_\theta L(z, \hat{\theta}) = 0
    \]

    \pause
    
    Rearranging:
    \[
        \eqnmarkbox[orange]{paramchange}{\hat{\theta}_{\epsilon,z} - \hat{\theta}} = -\epsilon\, \eqnmarkbox[blue]{invH}{H_{\hat{\theta}}^{-1}} \eqnmarkbox[red]{grad}{\nabla_\theta L(z, \hat{\theta})}
    \]
    \annotate[yshift=1em]{above}{invH}{\small curvature of the loss surface}
    \annotate[yshift=-1em]{below,left}{paramchange}{\small change in parameters}
    \annotate[yshift=-1em]{below,right}{grad}{\small direction of steepest ascent for $z$}

    \pause
    \vspace{0.5cm}
    \textbf{Step 4.} Formal definition of the \emph{influence function} is:
    \[
        \mathcal{I}_\theta(z) \;:=\; \left.\frac{d\hat{\theta}_{\epsilon,z}}{d\epsilon}\right|_{\epsilon=0} = -H_{\hat{\theta}}^{-1}\,\nabla_\theta L(z, \hat{\theta})
    \]
\end{frame}

\begin{frame}{Approximating Leave-One-Out}
    Removing $z$ $\Leftrightarrow$ setting $\epsilon = -\frac{1}{n}$.

    \vspace{0.3cm}
    \textbf{First-order approximation} using the influence function $\mathcal{I}_\theta(z)$:
    \begin{equation}
        \hat{\theta}_{-z} - \hat{\theta} \approx -\tfrac{1}{n}\,\mathcal{I}_\theta(z) = \frac{1}{n} H_{\hat{\theta}}^{-1} \nabla_{\theta} L(z, \hat{\theta})
    \end{equation}

    \pause
    \vspace{0.3cm}
    Closed-form LOO approximation:
    \begin{equation}
        \hat{\theta}_{-z} \approx \hat{\theta} - \tfrac{1}{n}\,\mathcal{I}_\theta(z) = \hat{\theta} + \frac{1}{n} H_{\hat{\theta}}^{-1} \nabla_{\theta} L(z, \hat{\theta})
    \end{equation}
\end{frame}


% Slide 5: Influence on Test Loss
\begin{frame}{Influence on Test Predictions}
    How does removing $z$ affect the loss on a test point $z_{\text{test}}$?

    \pause
    
    \vspace{0.3cm}
    \textbf{Chain rule} \citep{koh2017understanding}: remove point $\to$ change in parameters $\to$ change in prediction.
    \vspace{0.5cm}
    \begin{align*}
        \mathcal{I}_L(z, z_{\text{test}}) \;\;=\;\;
        \eqnmarkbox[purple]{prediff}{\frac{d\, L(z_{\text{test}}, \hat{\theta}_{\epsilon,z})}{d\epsilon}}
        &= \eqnmarkbox[orange]{testgrad}{\nabla_\theta L(z_{\text{test}}, \hat{\theta})^\top}
        \cdot
        \eqnmarkbox[teal]{paramdiff}{\frac{d\hat{\theta}_{\epsilon,z}}{d\epsilon}} \\[2.5em]
        &= -\, \nabla_{\theta} L(z_{\text{test}}, \hat{\theta})^\top\, H_{\hat{\theta}}^{-1}\, \nabla_{\theta} L(z, \hat{\theta})
    \end{align*}
    \annotate[yshift=0.8em]{above,left}{prediff}{\small change in prediction}
    \annotate[yshift=1.3em]{above}{testgrad}{\small test gradient}
    \annotate[yshift=-0.5em]{below,left}{paramdiff}{\small $\mathcal{I}_\theta(z)$: change in parameters}
\end{frame}

\begin{frame}{Generalizing to Any Evaluation Function}
    The test loss $L(z_{\text{test}}, \theta)$ is just one choice.
    We can replace it with \emph{any differentiable evaluation function} $f(\theta)$:

    \begin{equation}
        \mathcal{I}_{f}(z) = - \textcolor{red}{\nabla_{\theta} f(\hat{\theta})^\top} H_{\hat{\theta}}^{-1} \nabla_{\theta} L(z, \hat{\theta})
    \end{equation}

    \pause
    \vspace{0.3cm}
    \textbf{Examples:}
    \begin{itemize}\setlength\itemsep{0.6em}
        \item $f(\theta) = L(z_{\text{test}}, \theta)$: influence on a single test loss (previous slide)
        \item $f(\theta) = \frac{1}{|\mathcal{D}_{\text{val}}|}\sum_{z' \in \mathcal{D}_{\text{val}}} L(z', \theta)$: influence on validation loss
        \item $f(\theta) = L_{\text{safety}}(\mathcal{D}_{\text{safe}}, \theta)$: influence on safety alignment
        \item $f(\theta) = \|\theta\|^2$: influence on model complexity
    \end{itemize}
\end{frame}

\begin{frame}{Will It Work for Modern Deep Networks?}
    Three assumptions behind the classical derivation are \remph{violated} for deep networks:

    \vspace{0.4cm}
    \begin{itemize}\setlength\itemsep{2em}
        \item \textbf{Non-convexity.} The loss landscape is non-convex --- the Hessian $H_{\hat\theta}$ is often \remph{not positive semi-definite} \citep{basu2021influence}.

        \item \textbf{Non-convergence.} Networks are rarely trained to a true minimum (early stopping, finite epochs, optimizer noise) --- so $\nabla_\theta R(\hat\theta) \neq 0$, \remph{breaking the stationarity} condition in the derivation.
    \end{itemize}
\end{frame}


\begin{frame}{Fix 1: Damping --- Addressing Non-Convexity and Non-Convergence}
    \citet{koh2017understanding}: form a \textbf{convex quadratic approximation} around the current parameters $\tilde{\theta}$ (not necessarily a minimum):
x
    \begin{equation}
        \tilde{L}(z, \theta) = \eqnmarkbox[gray]{qconst}{L(z, \tilde{\theta})} + \eqnmarkbox[orange]{qlinear}{\nabla L(z, \tilde{\theta})^\top (\theta - \tilde{\theta})} + \frac{1}{2}(\theta - \tilde{\theta})^\top \eqnmarkbox[violet]{qdamp}{(H_{\tilde{\theta}} + \lambda I)} (\theta - \tilde{\theta})
    \end{equation}
    \annotate[yshift=-1em]{below,left}{qconst}{\small constant}
    \annotate[yshift=-1em]{below}{qlinear}{\small linear term (gradient)}
    \annotate[yshift=-2.5em]{below,right}{qdamp}{\small damped Hessian}

    \pause
    \vspace{1.3cm}
    \begin{itemize}\setlength\itemsep{0.4em}
        \item \textbf{Non-convexity fix:} damping \remph{$\lambda I$} forces $(H_{\tilde{\theta}} + \lambda I) \succ 0$ --- a convex bowl.
        \item \textbf{Non-convergence fix:} expanding around \emph{any} $\tilde{\theta}$ (not the minimum) sidesteps the assumption $\nabla R(\hat\theta) = 0$.
    \end{itemize}
\end{frame}


\begin{frame}{Fix 2: Gauss-Newton Hessian --- Addressing Efficiency}
    Under mild assumptions, the Hessian can be approximated by the \textbf{Gauss-Newton Hessian (GNH)}, which is positive semi-definite and much cheaper to compute:

    \vspace{0.5cm}
    \textbf{Gauss-Newton Hessian (GNH):} a PSD approximation built from first-order quantities.
    \vspace{1em}
    \begin{equation}
        G = \eqnmarkbox[red]{jac1}{J_{y\theta}^\top} \eqnmarkbox[blue]{lossH}{H_y} \eqnmarkbox[red]{jac2}{J_{y\theta}}
    \end{equation}
    \annotate[yshift=-1em]{below,left}{jac2}{\small Jacobian of network outputs w.r.t.\ parameters}
    \annotate[yshift=1em]{above}{lossH}{\small Hessian of loss w.r.t.\ outputs (e.g., logits)}

    \pause
    \vspace{2em}
    \begin{itemize}\setlength\itemsep{0.3em}
        \item $H_y$ is tiny ($c\times c$ for $c$ outputs, often diagonal for softmax + cross-entropy).
    \end{itemize}
\end{frame}

% Slide 10: The Real Question being Answered
\begin{frame}{What are Influence Functions Actually Computing?}
    Influence function is not entirely wrong!

    \vspace{0.5cm}

    Then, in which regime, does influence function work correctly?

    \vspace{0.5cm}
    
    $\Rightarrow$ We \remph{do not} retrain the model from scratch when we compute influence functions. 
    
    Instead, we solve a \textbf{proximal optimization problem} that penalizes deviation from the current parameters and predictions.
    
\end{frame}


\begin{frame}{The PBRF: Formal Definition \citep{bae2022if}}
    Given current parameters $\tilde{\theta}$, the PBRF response to upweighting $z$ by $\epsilon$ is:
    \vspace{.3cm}
    \[
        \theta^{\text{PBRF}}_{\epsilon, z}
        \;:=\;
        \arg\min_{\theta}\;
        \eqnmarkbox[blue]{breg}{\frac{1}{n}\sum_{i=1}^{n} D_{L^{(i)}}\!\bigl(f(\theta, x_i),\, f(\tilde{\theta}, x_i)\bigr)}
        \;+\;
        \eqnmarkbox[red]{upw}{\epsilon\,L(z, \theta)}
        \;+\;
        \eqnmarkbox[violet]{prox}{\tfrac{\lambda}{2}\|\theta - \tilde{\theta}\|^2}
    \]
    \annotate[yshift=-0.5em]{below,left}{breg}{\small \shortstack[l]{Bregman divergence \\ preserves predictions on other data}}
    \annotate[yshift=-0.5em]{below}{upw}{\small upweight $z$}
    \annotate[yshift=0.5em]{above,left}{prox}{\small proximity to $\tilde{\theta}$}

    \vspace{1.2cm}
    \pause
    \textbf{Key properties:}
    \begin{itemize}\setlength\itemsep{0.3em}
        \item Objective is \remph{strictly convex} in $\theta$ (for $\lambda > 0$).
        \item No assumption that $\tilde{\theta}$ is a stationary point.
        \item Taylor-expanding gives the same influence formula (GNH+damping).
    \end{itemize}
\end{frame}

\begin{frame}{The i.i.d.\ Assumption Breaks Down}
    Part 1 assumed training data are \emph{independent}:

    \vspace{0.3cm}
    \begin{center}
        Remove $z_i$ $\Rightarrow$ only $z_i$'s contribution changes.
    \end{center}

    \pause
    \vspace{0.5cm}
    In many real-world settings, data points are \remph{interconnected}:

    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Graphs:} nodes share information via edges (social networks, molecules, citation graphs)
        \item \textbf{Sequences:} tokens depend on context (language models)
        \item \textbf{Groups:} correlated subsets of data (demographic slices, data sources)
    \end{itemize}

    \pause
    \vspace{0.5cm}
    Modifying one element can \remph{cascade} through the structure, affecting many predictions simultaneously.
\end{frame}

\begin{frame}{GNNs: Message Passing on Graphs}
    \begin{columns}[c]
    \begin{column}{0.55\textwidth}
    \begin{center}
    \begin{tikzpicture}[
        scale=0.85,
        node/.style={circle, draw, thick, minimum size=8mm, font=\small},

        \draw[thick] (v) -- (b);

        \draw[thick] (a) -- (b);
        \draw[thick] (c) -- (d);

        % Messages
        \draw[msg, blue!70] (a) to[bend right=15] node[left, font=\scriptsize] {$m_a$} (v);
        \draw[msg, blue!70] (b) to[bend left=15] node[left, font=\scriptsize] {$m_b$} (v);
        \draw[msg, blue!70] (c) to[bend left=15] node[right, font=\scriptsize] {$m_c$} (v);

    \end{tikzpicture}
    \end{center}
    \end{column}
    \begin{column}{0.45\textwidth}
        Each node updates its embedding by \emph{aggregating messages} from neighbors:

        \vspace{0.3cm}
        $h_v^{(l+1)} = \sigma\!\left(\sum_{u \in \mathcal{N}(v)} W^{(l)} h_u^{(l)}\right)$

        \vspace{0.5cm}
        \textbf{Key insight:} removing node, edge, or subgraph changes:
        \begin{itemize}
            \item What $v$ ``sees'' (message)
            \item How $\theta$ is learned (parameter)
        \end{itemize}
    \end{column}
    \end{columns}
\end{frame}

\begin{frame}{What Can We Perturb in Structured Data?}
    Beyond removing a single data point $z$:

    \vspace{0.5cm}
    \begin{itemize}
        \item \textbf{Edge edits:} add or remove a connection $e = (u,v)$ in a graph
        \item \textbf{Node removal:} remove a node and all its edges
        \item \textbf{Group removal:} remove a correlated subset of data
        \item \textbf{Feature perturbation:} modify attributes that propagate through structure
    \end{itemize}

    \pause
    \vspace{0.5cm}
    We start with \textbf{edge edits in GNNs} --- the simplest structural perturbation, but one that the standard influence function framework cannot handle.

\end{frame}

\begin{frame}{Why Edge Influence Is Hard}
    An edge edit affects GNN predictions through \emph{two distinct mechanisms}:

    \vspace{0.5cm}
    \begin{columns}[c]
    \begin{column}{0.48\textwidth}
        \centering
        \textbf{1. Parameter Shift}\\
        \vspace{0.3cm}
        Retraining on the modified graph changes the optimal parameters $\theta$.

        \vspace{0.3cm}
        $\theta^* \to \theta^*_{-e}$

        \vspace{0.3cm}
        {\small (Same as standard IF)}
    \end{column}
    \pause
    \begin{column}{0.04\textwidth}
        \centering
        \Large $+$
    \end{column}
    \begin{column}{0.48\textwidth}
        \centering
        \textbf{2. Message Propagation}\\
        \vspace{0.3cm}
        Removing/adding an edge changes how information flows through the network.

        \vspace{0.3cm}
        $G \to G \setminus e$

        \vspace{0.3cm}
        {\small (Unique to graph data)}
    \end{column}
    \end{columns}

    \pause
    \vspace{0.8cm}
    \centering
    Prior work (GIF) only captures the first pathway $\Rightarrow$ poor correlation ($\sim$0.1).
\end{frame}

\begin{frame}{Influence Decomposition}
    The total influence of an edge $e$ on a prediction $f$ decomposes into \emph{two distinct pathways}:

    \vspace{1.2cm}
    \begin{equation}
        \eqnmarkbox[teal]{total}{\mathcal{I}(e, f)} \;=\; \eqnmarkbox[blue]{pshift}{\mathcal{I}_{\text{param}}(e, f)} \;+\; \eqnmarkbox[red]{mprop}{\mathcal{I}_{\text{msg}}(e, f)}
    \end{equation}
    \annotate[yshift=1em]{above}{total}{\small total influence of edge $e$ on prediction $f$}
    \annotate[yshift=-1em]{below,left}{pshift}{\small 1. Parameter Shift: edge changes parameters}
    \annotate[yshift=-3em]{below,left}{mprop}{\small 2. Message Propagation: edge changes message passing}

\end{frame}

\begin{frame}{1. Parameter Shift: Removing $e$ from Training}
    Influence of $e$ on $f$ via parameter update (analogous to standard IF):
    \vspace{2em}
    \begin{equation}
        \mathcal{I}_{\text{param}}(e, f) = \frac{1}{N}\,
        \eqnmarkbox[orange]{testg}{\nabla_\theta f^\top}\,
        \eqnmarkbox[blue]{invG}{G^{-1}}\,
        \eqnmarkbox[red]{deltaG}{\sum_v \!\left(\nabla_\theta L(h^G_v) - \nabla_\theta L(h^{G_{-e}}_v)\right)}
    \end{equation}
    \annotate[yshift=1em]{above,left}{testg}{\small test gradient}
    \annotate[yshift=1em]{above}{invG}{\small damped GNH inverse}
    \annotate[yshift=-1em]{below, left}{deltaG}{\small training-gradient shift caused by removing edge $e$}

    \vspace{1.5cm}
    \pause
    The last term replaces the single-point gradient $\nabla_\theta L(z)$: removing an edge changes \emph{every node's} gradient because message passing couples them.
\end{frame}

\begin{frame}{2.Message Propagation: Direct Topology Effect}
    Influence of $e$ on $f$ via inference-time message passing (no retraining):
    \vspace{2em}
    \begin{equation}
        \mathcal{I}_{\text{msg}}(e, f) =
        \eqnmarkbox[violet]{sign}{(2 \cdot \mathbf{1}[e \in E] - 1)}
        \!\left(
        \eqnmarkbox[teal]{deriv}{\tfrac{\partial f}{\partial A_{uv}} + \tfrac{\partial f}{\partial A_{vu}}}
        \right)
    \end{equation}
    \annotate[yshift=1em]{above,left}{sign}{\small $+1$ if $e$ is added, $-1$ if removed}
    \annotate[yshift=-1em]{below,left}{deriv}{\small symmetric sensitivity of $f$ to the adjacency entries of $e$}

    \vspace{1.5cm}
    \pause
    \begin{itemize}\setlength\itemsep{0.3em}
        \item \emph{No Hessian inverse required} --- a single backward pass through the GNN.
        \item Captures how $e$ reroutes messages between $u$ and $v$, holding $\theta$ fixed.
    \end{itemize}
\end{frame}

\begin{frame}{Two Pathologies of Message Passing}
    Stacking more GNN layers trades one failure mode for another.

    \vspace{0.8em}
    \begin{itemize}\setlength\itemsep{0.8em}
        \item \textbf{Over-squashing} \citep{alon2020bottleneck}: information from distant nodes must be compressed through \remph{narrow bottlenecks}. Long-range dependencies get lost.
        \pause
        \item \textbf{Over-smoothing} \citep{li2018deeper}: as messages propagate, node representations become \remph{indistinguishable} --- every node converges to the same embedding.
    \end{itemize}

    \pause
    \vspace{1em}
    \textbf{The dilemma:} adding edges between communities widens bottlenecks (helps over-squashing) but also mixes their representations (worsens over-smoothing). Whether a specific edge is beneficial or harmful is \emph{not obvious from topology alone}.
\end{frame}

\begin{frame}{Potential Applications}
    \textbf{1. Adversarial Attack \& Defense}
    \begin{itemize}
        \item Craft structural attacks by targeting the most harmful edges.
        \item Detect and prune adversarially inserted edges.
    \end{itemize}

    \vspace{0.3cm}
    \textbf{2. Principled Graph Rewiring}
    \begin{itemize}
        \item Evaluate whether a proposed rewiring actually helps the downstream task.
        \item Design rewirings that balance \emph{over-squashing} and \emph{over-smoothing}.
    \end{itemize}

    \vspace{0.3cm}
    \textbf{3. Graph Data Curation}
    \begin{itemize}
        \item Identify noisy or mislabeled edges in knowledge graphs and citation networks.
        \item Select the most informative edges to label under an active-learning budget.
    \end{itemize}

    \vspace{0.3cm}
    \textbf{4. Interpretability \& Fairness}
    \begin{itemize}
        \item Explain a GNN's prediction on a node by tracing it back to influential edges.
        \item Audit which relationships drive potentially biased predictions.
    \end{itemize}

\begin{frame}{Limitation of First-Order Influence}
    Let $g_f = \nabla_\theta f(\theta^*)$, $g_S = \nabla_\theta \mathcal{L}_S(\theta^*)$, $H = \nabla_\theta^2 \mathcal{L}(\theta^*)$.

    \vspace{0.3cm}
    First-order influence is \emph{linear} in the selected subset:
    \begin{equation}
        f(\theta_\varepsilon) - f(\theta^*) \approx -\varepsilon \, g_f^\top H^{-1} g_S
    \end{equation}

    \vspace{0.5cm}
    \textbf{Problem:} influence of a group $S = \{z_1, \dots, z_m\}$ is just the \remph{sum of individual influences}~\citep{koh2019accuracy}:
    \[
        \mathcal{I}(S) = \sum_{i \in S} \mathcal{I}(z_i)
    \]

    \pause
    \vspace{0.3cm}
    Cannot capture:
    \begin{itemize}
        \item \textbf{Redundancy:} two similar points provide diminishing returns
        \item \textbf{Complementarity:} two points together help more than either alone
        \item \textbf{Interference:} two points together hurt despite each being helpful
    \end{itemize}
\end{frame}


\begin{frame}{Second-Order Group Influence}
    Let $\bar{v} = H^{-1}g_S$, \ $u = H^{-1}g_f$, \ $H_f = \nabla_\theta^2 f(\theta^*)$, \ $H_S = \nabla_\theta^2 \mathcal{L}_S(\theta^*)$.

    \vspace{0.5cm}
    Include the $\varepsilon^2$ term to capture \emph{interactions}:

    \begin{equation}
        f(\theta_\varepsilon) - f(\theta^*) \approx
        \underbrace{-\varepsilon \, g_f^\top \bar{v}}_{\mathcal{I}_1 \text{: first-order}}
        + \frac{\varepsilon^2}{2}\bigg[
        \underbrace{\bar{v}^\top H_f \, \bar{v}}_{\mathcal{T}_1 \text{: interaction attribution}}
        + \underbrace{2\, u^\top H_S \bar{v}}_{\mathcal{T}_2 \text{: curvature correction}}
        \bigg]
    \end{equation}

    \vspace{0.5cm}
    \begin{itemize}
        \item $\mathcal{T}_1$: how the parameter shift interacts with \emph{evaluation} curvature
        \item $\mathcal{T}_2$: how the group's \emph{training} curvature deflects the update
    \end{itemize}
\end{frame}

\begin{frame}{$\mathcal{T}_1$: Pairwise Interaction Expansion}
    Let $v_i = H^{-1}\nabla_\theta \ell(z_i, \theta^*)$. Then $\mathcal{T}_1$ decomposes into pairwise terms:

    \begin{equation}
        \mathcal{T}_1 = \frac{1}{|S|^2}\sum_{i,j \in S} v_i^\top H_f \, v_j
    \end{equation}

    \vspace{0.5cm}
    Each $v_i^\top H_f v_j$ measures how data points $i$ and $j$ \emph{interact} under the evaluation curvature.

    \vspace{0.5cm}
    This enables:
    \begin{itemize}
        \item Detecting \textbf{redundancy}: $v_i^\top H_f v_j > 0$ (diminishing returns)
        \item Detecting \textbf{complementarity}: $v_i^\top H_f v_j < 0$ (synergy)
    \end{itemize}
\end{frame}

\begin{frame}{$\mathcal{T}_2$: Curvature Correction \citep{basu2020second}}
    $$\mathcal{T}_2 = 2\,u^\top H_S \,\bar{v}, \qquad u = H^{-1}g_f, \quad \bar{v} = H^{-1}g_S$$

    \vspace{0.5cm}
    $H_S$ rotates the update direction $\bar{v}$. The correction $u^\top (H_S \bar{v})$ asks: does this rotation \emph{help} or \emph{hurt} evaluation?

    \vspace{0.5cm}
    \begin{itemize}
        \item $\mathcal{T}_2 > 0$: the group's curvature deflects the update \remph{toward} the evaluation objective $\Rightarrow$ \emph{harmful} (increases loss)
        \item $\mathcal{T}_2 < 0$: deflection is \emph{away} from evaluation loss $\Rightarrow$ \emph{beneficial}
        \item $\mathcal{T}_2 \approx 0$: group curvature is orthogonal to evaluation --- no correction needed
    \end{itemize}

    \vspace{0.5cm}
    Large $\mathcal{T}_2$ signals that first-order influence \emph{misjudges} the group's effect.
\end{frame}

\begin{frame}{Greedy Data Selection}
    Let $w_i = H_f v_i$. Predicted gain when adding point $j$ to current set $\mathcal{S}$:

    \begin{equation}
        \Delta f(j) = \underbrace{-\frac{1}{N} g_f^\top v_j}_{\text{first-order}} + \underbrace{\frac{1}{2N^2}\left(2\sum_{i \in \mathcal{S}} w_j^\top v_i + w_j^\top v_j\right)}_{\text{interaction}}
    \end{equation}

    \vspace{0.5cm}
    \begin{itemize}
        \item Precompute $v_i, w_i$ for all pool data
        \item Each greedy step reduces to dot products
    \end{itemize}
\end{frame}
